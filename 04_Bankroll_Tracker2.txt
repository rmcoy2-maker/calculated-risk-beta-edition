def exports_dir() -> Path:
    env = os.environ.get("EDGE_EXPORTS_DIR", "").strip()
    if env:
        p = Path(env)
        p.mkdir(parents=True, exist_ok=True)
        return p

    here = Path(__file__).resolve()
    for up in [here.parent] + list(here.parents):
        if up.name.lower() == "edge-finder":
            p = up / "exports"
            p.mkdir(parents=True, exist_ok=True)
            return p

    p = Path.cwd() / "exports"
    p.mkdir(parents=True, exist_ok=True)
    return p


def latest_existing(paths: list[Path]) -> Path | None:
    existing = [p for p in paths if p.exists()]
    return max(existing, key=lambda p: p.stat().st_mtime) if existing else None


def pick_first_existing(columns: list[str], df: pd.DataFrame) -> str | None:
    for col in columns:
        if col in df.columns:
            return col
    return None


@st.cache_data(ttl=60)
def load_bankroll_data() -> tuple[pd.DataFrame, pd.DataFrame, dict[str, Path | None]]:
    exp = exports_dir()

    bankroll_candidates = [
        exp / "bankroll.csv",
        exp / "bankroll_history.csv",
        exp / "bankroll_tracker.csv",
    ]
    bets_candidates = [
        exp / "bets_log.csv",
        exp / "bets.csv",
        exp / "wagers.csv",
    ]

    bankroll_path = latest_existing(bankroll_candidates)
    bets_path = latest_existing(bets_candidates)

    bankroll = pd.read_csv(bankroll_path, low_memory=False) if bankroll_path else pd.DataFrame()
    bets = pd.read_csv(bets_path, low_memory=False) if bets_path else pd.DataFrame()

    return bankroll, bets, {"bankroll": bankroll_path, "bets": bets_path}

    ["bankroll", "balance", "ending_bankroll", "roll", "equity"],
    bankroll,
)

if date_col is None or value_col is None:
    st.error("Could not identify date and bankroll columns in bankroll file.")
    st.write("Columns found:", list(bankroll.columns))
    st.stop()

bankroll[date_col] = pd.to_datetime(bankroll[date_col], errors="coerce")
bankroll[value_col] = pd.to_numeric(bankroll[value_col], errors="coerce")
bankroll = bankroll.dropna(subset=[date_col, value_col]).sort_values(date_col).copy()

if bankroll.empty:
    st.warning("Bankroll file loaded, but no valid dated bankroll rows were found.")
    st.stop()

bankroll["daily_change"] = bankroll[value_col].diff()
bankroll["daily_change_pct"] = bankroll[value_col].pct_change()
bankroll["peak_bankroll"] = bankroll[value_col].cummax()
bankroll["drawdown"] = bankroll[value_col] - bankroll["peak_bankroll"]
bankroll["drawdown_pct"] = bankroll["drawdown"] / bankroll["peak_bankroll"]

start_bankroll = float(bankroll[value_col].iloc[0])

worst_drawdown_pct = float(bankroll["drawdown_pct"].min()) if bankroll["drawdown_pct"].notna().any() else 0.0
net_change = current_bankroll - start_bankroll

col2.metric("Net change", f"${net_change:,.2f}")
col3.metric("Peak bankroll", f"${peak_bankroll:,.2f}")
col4.metric("Worst drawdown", f"${worst_drawdown:,.2f}", f"{worst_drawdown_pct:.1%}")

            bets.dropna(subset=[bets_date_col])
            .assign(activity_date=lambda d: d[bets_date_col].dt.normalize())

            .agg(
                bets_count=(pnl_col, "size"),
                total_pnl=(pnl_col, "sum"),
