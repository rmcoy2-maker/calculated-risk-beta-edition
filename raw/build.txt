# ---------- discovery ----------
def find_play_files() -> list[Path]:
    found: list[Path] = []
    seen: set[Path] = set()
    for base in DATA_DIRS:
        if not base.exists():
            continue
        for path in sorted(base.glob("*_plays.csv")):
            resolved = path.resolve()
            if resolved not in seen:
                seen.add(resolved)
                found.append(path)
    return sorted(found, key=lambda p: int(re.search(r"(\d{4})_plays\.csv$", p.name).group(1)))


# ---------- normalization helpers ----------
def slugify(value: object) -> str:
    text = "" if pd.isna(value) else str(value)
    text = text.strip().lower()
    text = re.sub(r"[^a-z0-9]+", "_", text)
    return text.strip("_")


def normalize_team(value: object) -> str:
    if pd.isna(value):
        return ""
    text = str(value).strip()
    return TEAM_NORMALIZATION.get(text, text)


def normalize_possession_team(value: object) -> str:
    text = normalize_team(value)
    # If it is still a full city/team string that was not caught above,
    # fall back to the final token except for 49ers.
    if not text:
        return text
    if text in TEAM_NORMALIZATION.values():
        return text
    if text.endswith("49ers"):
        return "49ers"
    tokens = text.split()
    return tokens[-1] if tokens else text


def parse_date_mmdd_with_season(date_value: object, season_value: object) -> pd.Timestamp:
    if pd.isna(date_value) or pd.isna(season_value):
        return pd.NaT

    raw = str(date_value).strip()
    season = int(season_value)

    for fmt in ("%m/%d/%Y", "%Y-%m-%d", "%m/%d"):
        try:
            if fmt == "%m/%d":
                dt = pd.to_datetime(f"{raw}/{season}", format="%m/%d/%Y", errors="raise")
                # NFL regular/postseason spills into next calendar year.
                if dt.month in (1, 2):
                    dt = dt.replace(year=season + 1)
                return dt
            return pd.to_datetime(raw, format=fmt, errors="raise")
        except Exception:
            continue

    parsed = pd.to_datetime(raw, errors="coerce")
    if pd.notna(parsed) and parsed.year == 1900:
        parsed = parsed.replace(year=season)
    if pd.notna(parsed) and parsed.month in (1, 2) and parsed.year == season:
        parsed = parsed.replace(year=season + 1)
    return parsed


def week_sort_key(value: object) -> float:
    if pd.isna(value):
        return 999
    text = str(value).strip()
    if text in WEEK_ORDER:
        return WEEK_ORDER[text]
    match = re.search(r"(\d+)", text)
    if match:
        return float(match.group(1))
    return 999


def season_type(value: object) -> str:
    if pd.isna(value):
        return "unknown"
    text = str(value).strip().lower()
    if "hall of fame" in text or "preseason" in text:
        return "preseason"
    if text in {"wild card", "divisional", "conference", "super bowl"}:
        return "postseason"
    return "regular"


# ---------- transform ----------
def load_and_standardize(path: Path) -> pd.DataFrame:
    df = pd.read_csv(path, low_memory=False)

    missing = [c for c in PLAY_COLUMNS if c not in df.columns]
    if missing:
        raise ValueError(f"{path.name} is missing required columns: {missing}")

    df = df[PLAY_COLUMNS].copy()
    df["source_file"] = path.name

    df["Season"] = pd.to_numeric(df["Season"], errors="coerce").astype("Int64")
    df["Week"] = df["Week"].astype(str).str.strip()
    df["AwayTeam"] = df["AwayTeam"].map(normalize_team)
    df["HomeTeam"] = df["HomeTeam"].map(normalize_team)
    df["TeamWithPossession"] = df["TeamWithPossession"].map(normalize_possession_team)

    df["game_date"] = [parse_date_mmdd_with_season(d, s) for d, s in zip(df["Date"], df["Season"])]
    df["game_date_str"] = df["game_date"].dt.strftime("%Y-%m-%d")
    df["season_type"] = df["Week"].map(season_type)
    df["week_sort"] = df["Week"].map(week_sort_key)

    df["game_id"] = (
        df["Season"].astype(str)
        + "_"
        + df["game_date_str"].fillna("unknown_date")
        + "_"
        + df["AwayTeam"].map(slugify)
        + "_at_"
        + df["HomeTeam"].map(slugify)
    )

    df["play_sequence"] = df.groupby("game_id").cumcount() + 1
    df["drive_id"] = df["game_id"] + "_d" + df["DriveNumber"].fillna(-1).astype(int).astype(str)

    df["is_home_possession"] = (df["TeamWithPossession"] == df["HomeTeam"]).astype("Int64")
    df["is_away_possession"] = (df["TeamWithPossession"] == df["AwayTeam"]).astype("Int64")

    return df


# ---------- main ----------
def main() -> int:
    play_files = find_play_files()
    if not play_files:
        print("No *_plays.csv files found in project root, data/, exports/, or /mnt/data.")
        return 1

    print("Found play files:")
    for p in play_files:
        print(f"  - {p}")

    frames = [load_and_standardize(path) for path in play_files]
    plays = pd.concat(frames, ignore_index=True)

    plays = plays.sort_values(
        by=["Season", "week_sort", "game_date", "AwayTeam", "HomeTeam", "DriveNumber", "PlayNumberInDrive", "play_sequence"],
        kind="stable",
    ).reset_index(drop=True)

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    plays.to_csv(OUTPUT_PATH, index=False)

    summary = (
        plays.groupby(["Season", "season_type"], dropna=False)
        .agg(
            games=("game_id", "nunique"),
            plays=("game_id", "size"),
        )
        .reset_index()
    )

    print("\nWrote:", OUTPUT_PATH)
    print("Rows:", f"{len(plays):,}")
    print("Games:", f"{plays['game_id'].nunique():,}")
    print("Seasons:", f"{plays['Season'].nunique():,}")
    print("\nSeason summary:")
    print(summary.to_string(index=False))

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
